import cv2

cap = cv2.VideoCapture('video.mp4')
frames = []
while True:
    ret, frame = cap.read()
    if not ret:
        break
    frames.append(frame)
cap.release()

height, width = frames[0].shape[:2]
out = cv2.VideoWriter('reversed.mp4', cv2.VideoWriter_fourcc(*'mp4v'), 30, (width, height))

for frame in reversed(frames):
    out.write(frame)
    cv2.imshow('Reversed Video', frame)
    if cv2.waitKey(30) & 0xFF == ord('q'):
        break

out.release()
cv2.destroyAllWindows()
