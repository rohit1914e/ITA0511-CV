import cv2

def reverse_slow_video():
    video_path = r"C:\CV EXP\video.mp4"
    cap = cv2.VideoCapture(video_path)
    if not cap.isOpened():
        print("Video not found!")
        return
    frames = []
    while True:
        ret, frame = cap.read()
        if not ret:
            break
        frames.append(frame)
    cap.release()
    for frame in reversed(frames):
        cv2.imshow("Reverse Slow Motion", frame)
        if cv2.waitKey(150) & 0xFF == ord('q'):
            break
    cv2.destroyAllWindows()

reverse_slow_video()
